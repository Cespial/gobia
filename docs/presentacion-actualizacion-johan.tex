% ── 5. SPRINT C: LEY 617 ──
\begin{frame}{Sprint C: Ley 617 milim\'etrica}
\vspace{-4pt}
\scriptsize
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{\textcolor{rojo}{Antes (incorrecto)}}
\begin{itemize}\setlength{\itemsep}{0pt}
    \item ICLD filtraba por \textbf{nombre} de fuente
    \item Deducci\'on: 3\% autom\'atico fijo
    \item GF: cualquier fuente contaba
    \item Deducibles: sin condiciones
    \item Concejo: umbral \$1B fijo
    \item Vigencia: solo ``actual''
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{\textcolor{verde}{Ahora (correcto)}}
\begin{itemize}\setlength{\itemsep}{0pt}
    \item ICLD filtra por \textbf{c\'odigo} de fuente\\
    (1.2.1.0.00 y 1.2.4.3.04)
    \item Deducci\'on: \textbf{dato reportado real}\\
    + comparaci\'on vs calculado
    \item GF: \textbf{solo} fuente ICLD/SGP-LD
    \item Deducibles: condicionales\\
    (2.1.2.02.02.007 solo si fuente $\neq$ ICLD)
    \item Concejo: umbral 1000$\times$SMLMV\\
    (\$1.42B, indexado al salario m\'inimo)
    \item Vigencia: actual \textbf{+ futuras} (cod 1+4)
\end{itemize}
\end{column}
\end{columns}
\vspace{6pt}
\centering
\textcolor{sepia}{\scriptsize Resultado: indicador SI.17 ahora se calcula igual que la Contralor\'ia}
\end{frame}
